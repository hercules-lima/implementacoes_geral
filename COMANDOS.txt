26- Selecione Nome do Livro e Nome da Editora
SOLUÇÃO 1
SELECT LIVRO.TITULO, EDITORA.NOME
FROM LIVRO
INNER JOIN EDITORA ON (LIVRO.EDITORA_ID = EDITORA.ID);

SOLUÇÃO 2
SELECT LIVRO.TITULO, EDITORA.NOME
FROM LIVRO, EDITORA
WHERE LIVRO.EDITORA_ID = EDITORA.ID;
